public abstract class AbstractFactory
{
   public abstract GameObject CreateProduct();
}



public class EnemyMeleeFactory : AbstractFactory
    {
        private GameObject enemy;
        public override GameObject CreateProduct()
        {
            var newEnemyMelee = Resources.Load<GameObject>("Swordfish");
            return newEnemyMelee;
        }
    }


public class Spawner : MonoBehaviour
    {
        private void Start()
        {
            SpawnEnemyMelee();
        }

        public void SpawnEnemyMelee()
        {
            var spawnEnemy = new EnemyMeleeFactory();
            GameObject enemyMelee = spawnEnemy.CreateProduct();
            Instantiate(enemyMelee, transform.position, transform.rotation);
        }
    }
